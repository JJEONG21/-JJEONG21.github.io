import sys
sys.stdin = open('input.txt')

T = int(input())

def solve(i):
    global temp_total, total
    if i == N:
        if temp_total < total:
            total = temp_total
        return
    for j in range(N):
        if not visited_col[j]:
            visited_col[j] = 1
            temp_total += cards[i][j]
            solve(i+1)
            visited_col[j] = 0
            temp_total -= cards[i][j]





for tc in range(1, T+1):
    N = int(input())
    cards = [list(map(int, input().split())) for _ in range(N)]
    visited_col = [0]*N
    total = 99999
    temp_total = 0
    solve(0)
    print('#{} {}'.format(tc, total))
